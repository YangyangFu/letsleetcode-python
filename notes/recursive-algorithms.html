def maximum_recursion(A, i):
    if i < 1: 
        return A[0]
    else:
        return max(maximum_recursion(A, i-1),A[i])

def reverse(s):
    """Return a string in reverse order"""
    res = ''
    n = len(s)

    if n <= 1: # base case
        res = s
    else:
        r = reverse(s[:n-1])
        res = s[n-1]+r
    return res

print(reverse("hello world"))

    if n <= 1:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

# using recurrence - good recurrence using linear recursion - O(n)
def fibonacci(n):
    if n <= 1:
        return (n, 0)
    else:
        (a, b) = fibonacci(n-1)
        return (a+b, a)

print(fibonacci(100))

# using memoized recurrence - top-down dynamic programming O(n)
def fibonacci(n, r = {}):
    if n in r:
        return r[n]
    elif n <= 1:
        r[n] = n
    else:
        r[n] = fibonacci(n-1, r) + fibonacci(n-2, r)

    return r[n]

print(fibonacci(100))

# using bottom-up dynamic programming O(n)
def fibonacci(n):
    r = [0.]*(n+1)
    for j in range(n+1):
        if j <= 1:
            r[j] = j
        else:
            r[j] = r[j-1] + r[j-2]
    return r[n]

print(fibonacci(100))
